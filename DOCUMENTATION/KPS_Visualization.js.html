// App.js
import React, { useState } from 'react';
import Canvas from './KPS_Canvas';
import Button from './Button';
import convexHull from './KPS_Algo';
import './kpm.css';

/* These lines of code are declaring and initializing variables in the JavaScript file `App.js` for the
Convex Hull Visualization application. Here is what each variable is used for: */
let check = -1;
let fhull = [];
let noOfStates = 21;
let done = 0;

/**
 * The `App` function in JavaScript sets up a visualization for generating and displaying a convex hull
 * based on user input and random points.
 * @returns The `App` component is being returned. It contains the JSX structure for the Convex Hull
 * Visualization application, including the title, canvas container, button container, and description
 * container. The component also includes various state variables, functions for handling point clicks,
 * generating random points, generating convex hull, running convex hull, and updating the description.
 */
function App() {
  const [points, setPoints] = useState([]);
  const [convexHullPoints, setConvexHullPoints] = useState([]);
  // const [medianLine, setMedianLine] = useState(null); // State to hold the median line
  const [currentState, setCurrentState] = useState(-1);
  const [myArr, setMyArr] = useState([]);
  const [description, setDescription] = useState("You can see the Description for each state here.");

  // let noOfStates = 0;
/**
 * The function `handlePointClick` adds a new point to an array of points.
 * @param newPoint - The `newPoint` parameter in the `handlePointClick` function represents a new point
 * that is being added to an array of points.
 */
  const handlePointClick = (newPoint) => {
    setPoints([...points, newPoint]);
    // storePoints = [...points,newPoint];
  };

/**
 * The updateDescription function updates the description with a new value.
 * @param newDescription - The `updateDescription` function takes a `newDescription` parameter, which
 * is used to update the description in the code.
 */
  const updateDescription = (newDescription) => {
    setDescription(newDescription);
  };

 /**
  * The function `generateRandomPoints` generates 20 random points with x and y coordinates within
  * specified canvas dimensions and adds them to an existing array of points.
  */
  const generateRandomPoints = () => {
    const randomPoints = [];
    for (let i = 0; i < 20; i++) {
      // Generate random x and y coordinates within canvas dimensions
      const randomX = Math.random() * 500; // Change canvasWidth to your canvas width
      const randomY = Math.random() * 500; // Change canvasHeight to your canvas height
      randomPoints.push({ x: randomX, y: randomY });
    }
    // points.push(...randomPoints);
    setPoints([...points, ...randomPoints]);
  };

/**
 * The function `generateConvexHull` calculates the convex hull of a set of points and updates the
 * convex hull points and an array with the result.
 */
  const generateConvexHull = () => {
    done = 0;
    check = -1;
    fhull = convexHull(points);
    console.log(points.length);
    console.log(fhull.length);
    setConvexHullPoints(fhull[fhull.length - 1][2]);
    setMyArr(fhull);
  };

/**
 * The `runConvexHull` function runs a continuous loop at intervals of 600 milliseconds until a
 * condition is met.
 */
  const runConvexHull = () => {
    const interval = setInterval(() => {
      if (!done) {
        nextState();
      }
      else {
        clearInterval(interval);
      }
    }, 600)
  };

/**
 * The `nextState` function increments the `check` variable and sets the current state if `check` is
 * less than `noOfStates - 1`, otherwise it sets `done` to 1.
 */
  const nextState = () => {
    // document.getElementById("description-container").innerHTML = "Description for";
    if (check < noOfStates - 1) {
      check = check + 1;
      setCurrentState(check);
    }
    else if (check === noOfStates - 1) {
      done = 1;
    }
  };

/**
 * The `prevState` function decreases the `check` variable by 1 and updates the current state if
 * `check` is greater than 0.
 */
  const prevState = () => {
    done = 0;
    if (check > 0) {
      check = check - 1;
      setCurrentState(check);
    }
  };

  return (
    <div className="App">
      <h1>Convex Hull Visualization</h1>
      <div className="canvas-container">
        <Canvas points={points} convexHullPoints={convexHullPoints} onClick={handlePointClick} currentState={currentState} myArr={myArr} updateDescription={updateDescription} />
      </div>
      <div className="button-container">
        <Button onClick={generateConvexHull} label="Generate Convex Hull" />
        <Button onClick={nextState} label="Next State" />
        <Button onClick={prevState} label="Previous State" />
        <Button onClick={runConvexHull} label="RunConvexHull" />
        <Button onClick={generateRandomPoints} label="Generate Random Points" />
      </div>
      <div className="description-container">
          <p>{description}</p>
      </div>
    </div>
  );
}
