 */

  };

/**
 * The function `drawVerticalLine` draws a vertical line on a canvas context at a specified
 * x-coordinate with a specified color.
 * @param ctx - The `ctx` parameter is the 2D drawing context of the canvas element. It is used to draw
 * on the canvas using various drawing methods and properties.
 * @param x - The `x` parameter in the `drawVerticalLine` function represents the x-coordinate where
 * you want to draw the vertical line on the canvas.
 * @param canvasHeight - The `canvasHeight` parameter represents the height of the canvas on which you
 * are drawing the vertical line. It is used to determine the length of the vertical line that will be
 * drawn from the top to the bottom of the canvas at the specified x-coordinate.
 * @param color - The `color` parameter in the `drawVerticalLine` function is a string that represents
 * the color you want to use for drawing the vertical line on the canvas. It can be specified using
 * color names (e.g., "red", "blue", "green") or using hexadecimal color codes (e
 */
  const drawVerticalLine = (ctx, x, canvasHeight, color) => {
    ctx.beginPath();
    ctx.strokeStyle = color;
    ctx.moveTo(x, 0);
    ctx.lineTo(x, canvasHeight);
    ctx.stroke();
  };
  const canvasStyle = {
    border: '1px solid black',
    backgroundColor: 'rgba(255, 255, 255, 1)' // White color with full opacity
  };

  return (
    <canvas
      ref={canvasRef}
      width={500}
      height={500}
      onClick={handleCanvasClick}
      style={canvasStyle}
    ></canvas>
  );
};

 * such as the coordinates of the mouse pointer at the time of the click.
 */
const handleCanvasClick = (event) => {
  const canvas = canvasRef.current;
  const rect = canvas.getBoundingClientRect();
  const x = event.clientX - rect.left;
  const y = event.clientY - rect.top;
  onClick({ x, y });
};

*/
const drawVerticalLine = (ctx, x, canvasHeight, color) => {
  ctx.beginPath();
  ctx.strokeStyle = color;
  ctx.moveTo(x, 0);
  ctx.lineTo(x, canvasHeight);
  ctx.stroke();
};
